// Creating user table
Table users as U {
  u_id int [pk, increment] // auto-increment
  user_name varchar
  user_state varchar
  user_birth_date timestamp
  deviceID varchar
}

// Creating business table
Table business as B{
  b_id int [pk, increment]
  business_id varchar
  name varchar
  address varchar
  city varchar
  state varchar
  postal_code int
  latitude float
  longitude float
  review_count int
  categories varchar
 }
 
 // Creating user scans table
 Table scans as S {
  id int [pk] 
  scan_time timestamp
  u_id int
  b_id int
}

// Creating references
// You can also define relaionship separately
// > many-to-one; < one-to-many; - one-to-one
Ref: U.u_id < S.u_id  
Ref: B.b_id < S.b_id
